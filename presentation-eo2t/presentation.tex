% Xs Slide 3: Introduction + Rationale + What will be done pt.1 (just topic)
\begin{frame}
    \frametitle{Introduction: Thesis objectives and rationale}
    \begin{itemize}
        \item \textbf{Objective}:
            \begin{itemize}
                \item research and analyze existing terminal assistant solutions
                    using LLM.
                \item \textbf{design and implement} a CLI/TUI terminal assistant
                    based on local LLM.
                \item \textbf{evaluate} the effectiveness of the proposed
                    solution.
            \end{itemize}
        \item \textbf{Limitations for local execution}: usage of local LLMs
            under 2B parameters (suitable for devices with 8GB RAM)
    \vspace{0.5cm} \small
        \item \textbf{Rationale}: offline execution, independence from remote
            servers, network access, privacy
    \end{itemize}
\end{frame}

% Xs Slide 5: What will be done pt.2 (approaches desc)
\begin{frame}
    \frametitle{Initial design}
    The application is designed to be an interactive CLI shell, able to execute shell
    commands directly and answer natural language queries. It will consist of
    following components:
    \begin{itemize}
        \item \textbf{Shell Integration \& Classifier}: Bash vs Natural Language Query, LLM-based
        \item \textbf{Agentic}: LLM agent naive approach, LLM-based
        \item \textbf{RAG}: Document database with vector embeddings with LLM
            summarization, LLM-based and embedding-LLM-based
    \end{itemize}
    % NOTE: instead of focusing on NLQ to command generation, which is binded to
    % the LLM model (defined by its training data), we have decided to focus
    % more on more dynamic knowledge sources, which are represented by RAG,
    % while LLM does decision and summarization work
\end{frame}

\begin{frame}
    \frametitle{Shell Integration \& Classifier}
    \begin{multicols}{2}

    \begin{itemize}
        \item \textbf{Idea}: build a shell wrapper, which will behave either as a
            chat or as a normal shell, depending on the user query, using an
            LLM-based classifier.
        \item \textbf{Testing}: \texttt{classifier-100} contains 100 inputs with
            a balanced mix of natural language queries and shell command inputs
            (e.g. ``How do I list files in a directory?'', ``ls -la /home/user'')
    \end{itemize}

% Xs Slide 10: RAG pt.2 (results+conclusion)
\begin{frame}
    \frametitle{RAG approach}
    \begin{itemize}
        \item \textbf{Outcomes}:
            \begin{itemize}
                \item Robust precision and MRR with for all approaches
                \item Generated output size: second and third approaches
                    allowed to reduce and control output sizes
                \item Latency: The first and third approaches had lower
                    retrieval latency (avg. 0.5--1.5s) than the second approach
                    (avg. 2--3.5s), due to in-runtime reranking. Document-level
                    indexing cause high latencies when embedding big files.
            \end{itemize}
    \end{itemize}
\end{frame}

% Xs Slide 12: Latency tests pt.2 (results+conclusion)
\begin{frame}
    \frametitle{End-to-end latency tests}
    \begin{itemize}
        \item \textbf{Key Outcomes \& Latency Trade-offs}:
        \begin{itemize}
            \item \textbf{Smaller Embedding Models} (e.g., \texttt{granite-embedding:30m}):
                \begin{itemize}
                    \item \textbf{Document-level retrieval} delivers the best latency.
                    \item \textbf{Runtime passage reranking} allows to make generated
                context much smaller with similar precision.
                \end{itemize}
            \item \textbf{Larger Embedding Models} (e.g., \texttt{qwen3-embedding:0.6b}):
                \begin{itemize}
                    \item \textbf{Passage-level retrieval} heavily outperforms other methods.
                    \item Full-document vectorization and runtime reranking become too computationally expensive.
                \end{itemize}
        \end{itemize}
    \end{itemize}
\end{frame}

% Xs Slide 13: Conclusion (recap)
\begin{frame}
    \frametitle{Conclusion}
    \begin{itemize}
        \item \textbf{Shell Integration \& Classifier}: not viable with chosen
            approach (LLM-based), difficult to implement as a generic wrapper
        \item \textbf{Agentic}: unreliable, causing frequent model mistakes and
            hallucinations. Reduced the role of the LLM to summarizer in RAG
            component.
        \item \textbf{RAG}: robust and reliable within chosen constraints
            \vspace{0.5cm}
        \item \textbf{Architectural advice}: to achieve practical, local
            LLM assistance, it  must be constrained to specific tasks like
            retrieval and summarization, rather than open-ended classification
            or reasoning.
    \end{itemize}
\end{frame}
