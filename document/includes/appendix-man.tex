\label{a:man}

\paragraph{Conventional man page groups}
\begin{enumerate}
  \item Executable programs or shell commands
  \item System calls (functions provided by the kernel)
  \item Library calls (functions within program libraries)
  \item Special files (usually found in \texttt{/dev})
  \item File formats and conventions, e.g. \texttt{/etc/passwd}
  \item Games
  \item Miscellaneous (including macro packages and conventions),
      e.g. \texttt{man(7), groff(7), man-pages(7)}
  \item System administration commands (usually only for root)
  \item Kernel routines [Non standard]
\end{enumerate}

\paragraph{Conventional man page section names}
\begin{itemize}
  \item NAME
  \item SYNOPSIS (usage)
  \item CONFIGURATION
  \item DESCRIPTION
  \item OPTIONS (flags)
  \item EXIT STATUS
  \item RETURN VALUE
  \item ERRORS
  \item ENVIRONMENT
  \item FILES
  \item VERSIONS
  \item STANDARDS
  \item NOTES
  \item BUGS
  \item EXAMPLE
  \item AUTHORS
  \item SEE ALSO
\end{itemize}
